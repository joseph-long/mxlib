172 template<typename wavefrontT>
173 void fraunhoferPropagator<wavefrontT>::setWavefrontSizePixels(int wfsPix)
174 {
175  //If no change in size, do nothing
176  if(wfsPix == centerFocal.rows()) return;
177 
178  wavefrontSizePixels = wfsPix;
179 
180  xcen = 0.5*(wfsPix - 1.0);
181  ycen = 0.5*(wfsPix - 1.0);
182 
183  makeShiftPhase();
184 
185  fft_fwd.plan(wfsPix, wfsPix);
186 
187  fft_back.plan(wfsPix, wfsPix, MXFFT_BACKWARD);
188 }

190 template<typename wavefrontT>
191 void fraunhoferPropagator<wavefrontT>::makeShiftPhase()
192 {
193  realT pi = boost::math::constants::pi<realT>();
194 
195  //The normalization is included in the tilt.
196  realT norm = 1./(wavefrontSizePixels*sqrt(2));
197  complexT cnorm = complexT(norm, norm);
198 
199  //Resize the center phases
200  centerFocal.resize(wavefrontSizePixels, wavefrontSizePixels);
201  centerPupil.resize(wavefrontSizePixels, wavefrontSizePixels);
202 
203  //Shift by 0.5 pixels
204  realT arg = -2.0*pi*0.5*wavefrontSizePixels/(wavefrontSizePixels-1);
205 
206  for(int ii=0; ii < wavefrontSizePixels; ++ii)
207  {
208  for(int jj=0; jj < wavefrontSizePixels; ++jj)
209  {
210  centerFocal(ii,jj) = cnorm*exp(complexT(0.,arg*((ii-xcen)+(jj-ycen))));
211  centerPupil(ii,jj) = cnorm*exp(complexT(0., 0.5*pi - arg*((ii-xcen)+(jj-ycen))));
212  }
213  }
214 }
