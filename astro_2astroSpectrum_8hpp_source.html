77  ///Calculate the mean value of the spectrum when mutiplied by another.
78  /** The result is normalized by the mean value of the input spectrum, equivalent to:
79  * \f[
80  \mu = \frac{\int T(\lambda)S(\lambda) d\lambda}{\int T(\lambda) d\lambda}
81  \f]
82  * For instance, use this to get the mean value of a spectrum in a filter.
83  *
84  */
85  template<typename compSpectrumT>
86  realT mean( const compSpectrumT & T /**< [in] the spectrum to multiply by*/ )
87  {
88  realT sum1 = 0, sum2 = 0;
89 
90  for(int i=0; i< _spectrum.size(); ++i)
91  {
92  sum1 += _spectrum[i]*T[i];
93  sum2 += T[i];
94  }
95 
96  return sum1/sum2;
97  }

101 ///Class to manage an astronomical spectrum.
102 /** Details of the spectra, including units, file-location, and how they are read from disk, are specified by template parameter _spectrumT:
103  * see \ref astrophot_spectra "the available spectrum definitions."
104  * Spectra are loaded and immediately interpolated onto a wavelength grid. This is to facilitate manipulation of spectra, i.e.
105  * for multiplying by filters, etc.
106  *
107  * Inherits functions and operators to manipulate and characterize spectra from \ref mx::astro::baseSpectrum "baseSpectrum".
108  *
109  * \tparam _spectrumT is the underlying spectrum type, which provides (through static interfaces) the specifications of how to read or calculate the spectrum.
110  * \tparam freq specify whether this spectrum uses frequency instead of wavelength. Default is false.
111  *
112  * \ingroup astrophot
113  */

171  ///Load the spectrum and interpolate it onto a wavelength scale.
172  template<typename gridT>
173  int setSpectrum( gridT & lambda )
174  {
175  std::vector<realT> rawLambda;
176  std::vector<realT> rawSpectrum;
177 
178  if(_dataDir == "") _dataDir = ".";
179 
180  std::string path = _dataDir + "/" + spectrumT::fileName(_params);
181 
182  if( spectrumT::readSpectrum(rawLambda, rawSpectrum, path, _params) < 0)
183  {
184  return -1; ///\returns -1 on an error reading the spectrum.
185  }
186 
187  //Unit conversions
188  for(int i=0; i < rawLambda.size(); ++i)
189  {
190  rawLambda[i] /= spectrumT::wavelengthUnits/(units::length);
191  rawSpectrum[i] /= spectrumT::fluxUnits/(units::energy/(units::time * units::length * units::length * units::length));
192  }
193 
194  mx::gsl_interpolate(gsl_interp_linear, rawLambda, rawSpectrum, lambda, this->_spectrum);
195 
196  for(int i=0; i < lambda.size(); ++i)
197  {
198  if( !std::isnormal(this->_spectrum[i])) this->_spectrum[i] = 0;
199  }
200 
201  return 0; ///\returns 0 on success.
202  }
