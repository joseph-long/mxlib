151 ///Calculate an orthogonal projection of a basis on a pupil.
152 /** Can use either the stabilized Gramm Schmidt (SGS) procedure, or Singular Value Decomposition (SVD).
153  * This calls \ref applyPupil2Basis as a first step..
154  *
155  * \param [in] basisName the name of the basis to orthogonalize
156  * \param [in] pupilName the name of the pupil on which to orthogonalize.
157  * \param [in] method either \ref MXAO_ORTHO_METHOD_SGS (for SGS) or \ref MXAO_ORTHO_METHOD_SVD (for SVD)
158  *
159  * \tparam realT

161 template<typename realT>
162 void orthogonalizeBasis( const std::string & orthoName,
163  const std::string & basisName,
164  const std::string & pupilName,
165  int method )
166 {
167  fitsFile<realT> ff;
168  fitsHeader head;
169 
170 
171  eigenCube<realT> modes;
172  applyPupil2Basis(modes, basisName, pupilName);
173 
174  eigenCube<realT> ortho;
175  Eigen::Array<realT, -1, -1> spect;
176 
177  orthogonalizeBasis(ortho, spect, modes, method);
178 
179  modes.resize(1,1,1);
180 
181  std::string pupilFName = mx::AO::path::pupil::pupilFile(pupilName);
182  Eigen::Array<realT, -1, -1> pupil;
183  ff.read(pupilFName, pupil);
184 
185  realT psum = pupil.sum();
186  realT norm;
187 
188  for(int i=0; i< ortho.planes(); ++i)
189  {
190  norm = ortho.image(i).square().sum()/psum;
191 
192  ortho.image(i)/= sqrt(norm);
193 
194 
195  if(method == MXAO_ORTHO_METHOD_SGS)
196  {
197  spect.row(i) /= sqrt(norm);
198  }
199  }
200 
201  std::string orthoFName = mx::AO::path::basis::modes(orthoName, true);
202 
203  std::string orthm = "UNK";
204  if(method == MXAO_ORTHO_METHOD_SGS) orthm = "SGS";
205  if(method == MXAO_ORTHO_METHOD_SVD) orthm = "SVD";
206 
207  head.append("ORTHMETH", orthm, "Orthogonalization method.");
208  head.append("ORTHPUP", pupilName, "Pupil used for orthogonalization");
209 
210 
211  ff.write(orthoFName, ortho, head);
212 
213  if(method == MXAO_ORTHO_METHOD_SGS)
214  {
215  std::string orthoFName = mx::AO::path::basis::spectrum(orthoName, true);
216  ff.write(orthoFName, spect, head);
