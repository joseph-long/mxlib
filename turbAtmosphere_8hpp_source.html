258 int turbAtmosphere<realT>::genLayers()
259 {
260  if(_dataDir != "" && !_forceGen)
261  {
262 
263  std::string fbase = _dataDir;
264  fbase += "/";
265  fbase += "layer_";
266 
267  improc::fitsFile<realT> ff;
268  std::string fname;
269  for(int i=0; i< _layers.size(); ++i)
270  {
271  fname = fbase + mx::convertToString<int>(i) + ".fits";
272  ff.read(_layers[i].phase, fname);
273  }
274 
275  return 0;
276  }
277 
278  //#pragma omp parallel num_threads(2)
279  {
280  arrayT psd;
281  arrayT freq;
282  sigproc::psdFilter<realT> filt;
283 
284  mx::normDistT<realT> normVar;
285  normVar.seed();
286 
287  realT beta, L02, scrnSz, r0, L0, l0;
288 
289  realT sqrt_alpha = 0.5*11./3.;
290 
291  realT p;
292  // #pragma omp for
293  for(int i=0; i< _layers.size(); ++i)
294  {
295  std::cerr << "Generating layer " << i << " ";
296  scrnSz = _layers[i]._scrnSz;
297 
298  r0 = _layers[i]._r0;
299  L0 = _layers[i]._L0;
300  l0 = _layers[i]._l0;
301 
302  std::cerr << scrnSz << " " << r0 << " " << L0 << " " << l0 << "\n";
303  psd.resize(scrnSz, scrnSz);
304 
305  freq.resize(scrnSz, scrnSz);
306  sigproc::frequency_grid(freq, _pupD/_wfSz);
307 
308 
309  beta = 0.0218/pow( r0, 5./3.)/pow( _pupD/_wfSz,2) * pow(_lambda0/_lambda, 2);
310 
311  if(L0 > 0) L02 = 1.0/(L0*L0);
312  else L02 = 0;
313 
314 
315  for(int ii =0; ii < scrnSz; ++ii)
316  {
317  for(int jj=0; jj < scrnSz; ++jj)
318  {
319  if(freq(ii,jj) == 0 && L02 == 0)
320  {
321  p = 0;
322  }
323  else
324  {
325  p = beta / pow( pow(freq(ii,jj),2) + L02, sqrt_alpha);
326  if(l0 > 0 ) p *= exp(-1*pow( freq(ii,jj)*l0, 2));
327 
328  realT Ppiston = 0;
329  realT Ptiptilt = 0;
330  if(_subPiston)
331  {
332  Ppiston = pow(2*math::func::jinc(pi<realT>() * freq(ii,jj) * _pupD), 2);
333  }
334 
335  if(_subTipTilt)
336  {
337  Ptiptilt = pow(4*math::func::jinc2(pi<realT>() * freq(ii,jj) * _pupD), 2);
338  }
339 
340  p *= (1 - Ppiston - Ptiptilt);
341 
342  }
343  psd(ii,jj) = p;
344 
345  _layers[i].phase(ii,jj) = normVar;
346  }
347  }
348 
349  filt.psd(psd);
350 
351  filt(_layers[i].phase);
352  }
353  }//#pragma omp parallel
354 
355 
356  if(_dataDir != "")
357  {
358 
359  std::string fbase = _dataDir;
360  fbase += "/";
361  fbase += "layer_";
362 
363  improc::fitsFile<realT> ff;
364  std::string fname;
365  for(int i=0; i< _layers.size(); ++i)
366  {
367  fname = fbase + mx::convertToString<int>(i) + ".fits";
368  ff.write(fname, _layers[i].phase);
369  }
370  }
371 
372  return 0;

376 int turbAtmosphere<realT>::shift( arrayT & phase,
377  realT dt )
378 {
379  phase.resize(_wfSz, _wfSz);
380  phase.setZero();
381 
382  #pragma omp parallel for
383  for(int j=0; j< _layers.size(); ++j)
384  {
385  _layers[j].shift( dt );
386 
387  //This is faster than a separate loop by 5-10%
388  #pragma omp critical
389  phase += sqrt( _layers[j]._Cn2 ) * _layers[j].shiftPhase.block(_buffSz, _buffSz, _wfSz, _wfSz);
390  }

428 void turbAtmosphere<realT>::nextWF(wavefront<realT> & wf)
429 {
430 
431  int Npix = _pupil->sum();
432 
433  shift( wf.phase, _nWf * _timeStep);
434  ++_nWf;
435 
436  wf.phase = (wf.phase - (wf.phase* (*_pupil)).sum()/Npix)* (*_pupil);
