324 /// Convert a UTC timespec to TAI modified Julian date
325 /** Converts a timespec assumed to be in <a href="https://en.wikipedia.org/wiki/Coordinated_Universal_Time">Coordinated
326  * Universal Time (UTC)</a> to a <a href="https://en.wikipedia.org/wiki/Julian_day">
327  * Modified Julian Date (MJD)</a> in
328  * <a href="https://en.wikipedia.org/wiki/International_Atomic_Time">International Atomic Time (TAI)</a>.
329  *
330  * \param [out] djm the modified Julian day number
331  * \param [out] djmf the fraction of the day
332  * \param [in] tsp contains the UTC time
333  * \param [out] tm0 [optional] will be filled with the broken down UTC time
334  *
335  * \retval 1 SOFA dubious year [see SOFA documentation for iauDat]
336  * \retval 0 success
337  * \retval -1 SOFA bad year [see SOFA documentation for iauDat and iauCal2jd]
338  * \retval -2 SOFA bad month [see SOFA documentation for iauDat and iauCal2jd]
339  * \retval -3 SOFA bad day [see SOFA documentation for iauDat and iauCal2jd]
340  * \retval -4 SOFA bad fractional day [see SOFA documentation for iauDat and iauCal2jd]
341  * \retval -5 SOFA internal error [see SOFA documentation for iauDat and iauCal2jd]
342  * \retval -10 gmtime_r returned error, check errno
343  *
344  * \ingroup timeutils
345  */
346 inline
347 int timespecUTC2TAIMJD( double & djm, double & djmf, const timespec & tsp, tm * tm0)
348 {
349  double dat, djm0;
350  tm *tmrv;
351  int rv1, rv2;
352 
353  // Get the broken down time corresponding to tsp0
354  tmrv = gmtime_r(&tsp.tv_sec, tm0);
355  if(tmrv == 0) return -10;
356 
357  // Then determine deltaAT = TAI-UTC
358  rv1 = sofa::iauDat(1900+tm0->tm_year, 1+tm0->tm_mon, tm0->tm_mday, 0.0, &dat);
359  if(rv1 < 0) return rv1;
360 
361  // And get the MJD
362  rv2 = sofa::iauCal2jd(1900+tm0->tm_year, 1+tm0->tm_mon, tm0->tm_mday, &djm0, &djm);
363  if(rv2 < 0) return rv2;
364 
365  // Finally calculate the day fraction
366  djmf = ((double) tm0->tm_hour)/24.0 + ((double) tm0->tm_min)/(24.0*60.) + (((double) tm0->tm_sec)+((double) tsp.tv_nsec/1e9) + dat )/(24.0*3600.0);
367 
368  if(djmf >= 1.0)
369  {
370  djmf -= 1.0;
371  djm += 1.0;
372  }
373 
374  if(rv1) return rv1;
375  return 0;
376 }
