268 inline
269 int ds9Interface::spawn()
270 {
271  int i;
272  char resp[32];
273  char cmd[DS9_CMD_MAX_LENGTH];
274  int ret;
275 
276  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaaccess %s", _title.c_str());
277 
278  resp[0] = 0;
279  if( command_response(cmd, resp, 128) ) return -1;
280 
281  //Don't respawn if it already exists.
282  if(strcmp(resp, "yes\n") == 0) return 0;
283 
284  ret = system("xpans &");
285  if (WIFSIGNALED(ret))
286  {
287  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
288  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
289  }
290 
291  snprintf(cmd, DS9_CMD_MAX_LENGTH, "ds9 -title %s -port %i &", _title.c_str(), _port);
292 
293  ret = system(cmd);
294  if (WIFSIGNALED(ret))
295  {
296  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
297  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
298  }
299 
300  //Now wait for ds9 to respond or timeout.
301  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaaccess %s", _title.c_str());
302 
303  for(i=0;i<10000;i++)
304  {
305  resp[0] = 0;
306  if( command_response(cmd, resp, 128) ) return -1;
307  if(strcmp(resp, "yes\n") == 0) return 0;
308  usleep(100);
309  }
310 
311  return -1; //a timeout
312 }

358 int ds9Interface::display( const dataT * im,
359  size_t dim1,
360  size_t dim2,
361  size_t dim3,
362  int frame
363  )
364 {
365  size_t i, tot_size;
366  char cmd[DS9_CMD_MAX_LENGTH];
367  size_t pixsz;
368  int bitpix;
369  int ret;
370 
371  if(frame < 1)
372  {
373  fprintf(stderr, "frame must >= 1\n");
374  return -1;
375  }
376 
377  spawn();
378 
379  bitpix = getFitsBITPIX<dataT>();
380 
381  pixsz = sizeof(dataT);
382 
383  //If needed add a shared memory segment for this frame
384  if(frame-1 >= nsegs)
385  {
386  addsegment( frame );
387  }
388 
389  //Calculate total size
390  tot_size= pixsz;
391  tot_size*=dim1;
392  tot_size*=dim2;
393  tot_size*=dim3;
394 
395  //Re-allocate shared memory if necessary
396  if(tot_size > segs[frame-1].size)
397  {
398  if( segs[frame-1].size > 0 )
399  {
400  sharedmem_segment_detach( &segs[frame-1]);
401  }
402  sharedmem_segment_create( &segs[frame-1], tot_size);
403  }
404 
405  memcpy( segs[frame-1].addr, im, tot_size );
406 
407 
408  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s frame %i", _title.c_str(), frame);
409  ret = system(cmd);
410  if (WIFSIGNALED(ret))
411  {
412  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
413  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
414  }
415 
416 
417  //Handle single image so that the cube dialog doesn't open up if dim3=1
418  if(dim3 == 1)
419  {
420  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s shm array shmid %i [xdim=%zu,ydim=%zu,bitpix=%i] &",
421  _title.c_str(), segs[frame-1].shmemid,
422  dim1, dim2, bitpix);
423  }
424  else
425  {
426  snprintf(cmd, DS9_CMD_MAX_LENGTH, "xpaset -p %s shm array shmid %i [xdim=%zu,ydim=%zu,zdim=%zu,bitpix=%i] &",
427  _title.c_str(), segs[frame-1].shmemid,
428  dim1, dim2, dim3, bitpix);
429  }
430 
431  ret = system(cmd);
432  if (WIFSIGNALED(ret))
433  {
434  if( WTERMSIG(ret) == SIGINT) raise(SIGINT);
435  if( WTERMSIG(ret) == SIGQUIT) raise(SIGQUIT);
436  }
437 
438  return 0;
