33 /// Get the Zernike coefficients n,m corrresponding the Noll index j.
34 /** Calculates the single index j for (n,m) following Noll (1976) \cite noll_1976
35  * See also: http://en.wikipedia.org/wiki/Zernike_polynomials
36  *
37  * \param[out] n the radial index of the Zernike polynomial
38  * \param[out] m the azimuthal index of the Zernnike polynomial
39  * \param[in] j the Noll index, \f$j > 0\f$
40  *
41  * \retval 0 on success
42  * \retval -1 on error (j < 1)
43  */
44 int noll_nm( int & n, int & m, int j )
45 {
46  if( j < 1)
47  {
48  mxError("noll_nm", MXE_INVALIDARG, "The Noll index j cannot be less than 1 in the Zernike polynomials");
49  return -1;
50  }
51 
52 
53  n = ceil(-1.5 + sqrt(0.25 + 2*j) - 1e-10); // 1e-10 is to avoid wrong rounding due to numerical precision
54 
55  int jrem = j - (n * (n+1)/2+1);
56  m = (jrem + (jrem % 2) * abs( (n % 2)-1) + fabs( (jrem % 2)-1) * (n % 2)) * (-boost::math::sign( (j % 2)-0.5));
57 
58  return 0;
59 }

61 /// Calculate the coefficients of a Zernike radial polynomial
62 /**
63  * \param[out] c is allocated to length \f$ 0.5(n-m)+1\f$ and filled with the coefficients.
64  * \param[in] n is the radial index of the Zernike polynomial.
65  * \param[in] m is the azimuthal index of the Zernike polynomial.
66  *
67  * \retval 0 on success
68  * \retval -1 on error
69  *
70  * \tparam realT is a real floating type
71  */
72 template<typename realT>
73 int zernikeRCoeffs( std::vector<realT> & c, int n, int m)
74 {
75  m = abs(m);
76 
77  if( n < m)
78  {
79  mxError("zernikeRCoeffs", MXE_INVALIDARG, "n cannot be less than m in the Zernike polynomials");
80  return -1;
81  }
82 
83  //If odd, it's 0.
84  if( (n-m) % 2 > 0)
85  {
86  c.resize(1,0);
87  return 0;
88  }
89 
90  int ul = 0.5*(n-m) + 1;
91 
92  c.resize(ul);
93 
94  for(int k=0; k < ul; ++k)
95  {
96  c[k] = pow(-1.0, k) * boost::math::factorial<realT>(n - k) / ( boost::math::factorial<realT>(k) * boost::math::factorial<realT>(0.5*(n+m) - k)* boost::math::factorial<realT>(0.5*(n-m) - k));
97  }
98 
99  return 0;
100 }

116 {

257 ///Fill in an Eigen-like array with a Zernike polynomial
258 /** Sets any pixel which is at rad <= r < rad+0.5 pixels to rho = 1, to be consistent with mx::circularPupil
259  *
260  * \param[out] arr is the allocated array with an Eigen-like interface. The rows() and cols() members are used to size the polynomial.
261  * \param[in] n is the radial index of the polynomial
262  * \param[in] m is the azimuthal index of the polynomial
263  * \param[in] xcen is the x coordinate of the desired center of the polynomial, in pixels
264  * \param[in] ycen is the y coordinate of the desired center of the polynomial, in pixels
265  * \param[in] rad [optional] is the desired radius. If rad <= 0, then the maximum radius based on dimensions of m is used.
266  */
267 template<typename arrayT, int overscan=2>
268 int zernike( arrayT & arr,
269  int n,
270  int m,
271  typename arrayT::Scalar xcen,
272  typename arrayT::Scalar ycen,
273  typename arrayT::Scalar rad = -1 )
274 {
275  typename arrayT::Scalar x;
276  typename arrayT::Scalar y;
277  typename arrayT::Scalar r, rho;
278  typename arrayT::Scalar phi;
279 
280  std::vector<typename arrayT::Scalar> c;
281 
282  if(zernikeRCoeffs(c, n, m) < 0) return -1;
283 
284  size_t l0 = arr.rows();
285  size_t l1 = arr.cols();
286 
287  if(rad <= 0) rad = 0.5*std::min(l0-1, l1-1);
288 
289  for(size_t i=0; i < l0; ++i)
290  {
291  for(size_t j=0; j < l1; ++j)
292  {
293  x = i - xcen;
294  y = j - ycen;
295 
296 
297  r = std::sqrt( x*x + y*y );
298 
299  //This is to be consistent with mx::circularPupil while still respecting the Zernike rules
300  if(r > rad && r <= rad+(1.0/overscan)) r = rad;
301 
302  rho = r / rad;
303 
304  if(rho <= 1.0)
305  {
306  phi = std::atan2( y, x);
307  arr(i,j) = zernike(rho, phi, n, m, c);
308  }
309  else
310  {
311  arr(i,j) = 0.0;
312  }
313  }
314  }
315  return 0;
316 }
