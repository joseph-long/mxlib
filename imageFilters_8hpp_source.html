27 template<typename _arrayT, size_t _kernW=4>
28 struct gaussKernel
29 {
30  typedef _arrayT arrayT;
31  typedef typename _arrayT::Scalar arithT;
32  static const int kernW = _kernW;
33 
34  arrayT kernel;
35 
36  arithT _fwhm;
37 
38  gaussKernel(arithT fwhm)
39  {
40  _fwhm = fwhm;
41 
42  int w = kernW*_fwhm;
43 
44  if(w % 2 == 0) w++;
45 
46  kernel.resize(w, w);
47 
48  arithT kcen = 0.5*(w-1.0);
49 
50  arithT sig2 = _fwhm/2.354820045030949327;
51  sig2 *= sig2;
52 
53  arithT r2;
54  for(int i=0; i < w; ++i)
55  {
56  for(int j=0; j < w; ++j)
57  {
58  r2 = pow(i-kcen,2) + pow(j-kcen,2) ;
59  kernel(i,j) = exp(-r2/(2.0*sig2));
60  }
61  }
62 
63  kernel /= kernel.sum();
64  }
65 
66  int maxWidth()
67  {
68  return _kernW*_fwhm;
69  }
70 
71  void setKernel(arithT x, arithT y, arrayT & kernelArray)
72  {
73  kernelArray = kernel;
74  }

109  void setKernel(arithT x, arithT y, arrayT & kernel)
110  {
111  arithT rad = sqrt((arithT) (x*x + y*y));
112 
113  arithT sinq = y/rad;
114  arithT cosq = x/rad;
115 
116 
117  int w = 2*( (int) std::max( fabs(_azWidth*sinq), fabs(_radWidth*cosq) ) + 1 );
118  int h = 2*( (int) std::max( fabs(_azWidth*cosq), fabs(_radWidth*sinq) ) + 1);
119 
120 
121  kernel.resize(w, h);
122 
123  arithT xcen = 0.5*(w-1.0);
124  arithT ycen = 0.5*(h-1.0);
125 
126  arithT sinq2, cosq2, sindq;
127  for(int i=0; i < w; ++i)
128  {
129  for(int j=0; j < h; ++j)
130  {
131  rad = sqrt( pow(i-xcen,2) + pow(j-ycen,2) );
132  sinq2 = (j-ycen)/rad;
133  cosq2 = (i-xcen)/rad;
134  sindq = sinq2*cosq - cosq2*sinq;
135  if( rad <= _radWidth && fabs(rad*sindq) <= _azWidth) kernel(i,j) = 1;
136  else kernel(i,j) = 0;
137  }
138  }
139  if(kernel.sum() == 0)
140  {
141  std::cerr << "Kernel sum 0: " << x << " " << y << "\n";
142  exit(-1);
143  }
144  kernel /= kernel.sum();
145  }

188  * \tparam imageInT the type of the input image (must have an Eigen like interface)
189  * \tparam kernelT is the kernel type (see above)
190  *

192 template<typename imageOutT, typename imageInT, typename kernelT>
193 void filterImage(imageOutT & fim, imageInT im, kernelT kernel, int maxr= 0)
194 {
195  fim.resize(im.rows(), im.cols());
196 
197  float xcen = 0.5*(im.rows()-1);
198  float ycen = 0.5*(im.cols()-1);
199 
200  if(maxr == 0) maxr = 0.5*im.rows() - kernel.maxWidth();
201 
202  int mini = 0.5*im.rows() - maxr;
203  int maxi = 0.5*im.rows() + maxr;
204  int minj = 0.5*im.cols() - maxr;
205  int maxj = 0.5*im.cols() + maxr;
206 
207  typename kernelT::arrayT kernelArray;
208 
209  #pragma omp parallel private(kernelArray)
210  {
211  int im_i, im_j, im_p,im_q;
212  int kern_i, kern_j, kern_p,kern_q;
213  typename imageOutT::Scalar norm;
214 
215  #pragma omp for
216  for(size_t i=0; i< im.rows(); i++)
217  {
218  for(size_t j=0; j<im.cols(); j++)
219  {
220  if((i >= mini && i< maxi) && (j>= minj && j<maxj))
221  {
222  kernel.setKernel(i-xcen, j-ycen, kernelArray);
223  fim(i,j) = (im.block(i-0.5*(kernelArray.rows()-1), j-0.5*(kernelArray.cols()-1), kernelArray.rows(), kernelArray.cols())*kernelArray).sum();
224  }
225  else
226  {
227  kernel.setKernel(i-xcen, j-ycen, kernelArray);
228 
229  im_i = i - 0.5*(kernelArray.rows()-1);
230  if(im_i < 0) im_i = 0;
231 
232  im_j = j - 0.5*(kernelArray.cols()-1);
233  if(im_j < 0) im_j = 0;
234 
235  im_p = im.rows() - im_i;
236  if(im_p > kernelArray.rows()) im_p = kernelArray.rows();
237 
238  im_q = im.cols() - im_j;
239  if(im_q > kernelArray.cols()) im_q = kernelArray.cols();
240 
241  kern_i = 0.5*(kernelArray.rows()-1) - i;
242  if(kern_i < 0) kern_i = 0;
243 
244  kern_j = 0.5*(kernelArray.cols()-1) - j;
245  if(kern_j < 0) kern_j = 0;
246 
247  kern_p = kernelArray.rows() - kern_i;
248  if(kern_p > kernelArray.rows()) kern_p = kernelArray.rows();
249 
250  kern_q = kernelArray.cols() - kern_j;
251  if(kern_q > kernelArray.cols()) kern_q = kernelArray.cols();
252 
253  //Pick only the smallest widths
254  if(im_p < kern_p) kern_p = im_p;
255  if(im_q < kern_q) kern_q = im_q;
256 
257  norm = kernelArray.block(kern_i, kern_j, kern_p, kern_q ).sum();
258 
259  fim(i,j) = ( im.block(im_i, im_j, kern_p, kern_q) * kernelArray.block(kern_i, kern_j, kern_p, kern_q )).sum()/norm;
260 
261  if( !std::isfinite(fim(i,j))) fim(i,j) = 0.0;
262  }
263  }
264  }
265  }// pragma omp parallel

301 /** The radial profile is calculated using linear interpolation on a 1 pixel grid
302  *
303  * \param [out] radprof is the radial profile image. This will be resized.
304  * \param [in] im is the image to form the profile of.
305  * \param [in] rad is an array of radius values
306  * \param [in] subtract if true, then on ouput im will have had its radial profile subtracted.
307  *
308  * \tparam radprofT the eigen array type of the output
309  * \tparam eigenimT the eigen array type of the input
310  */
311 template<typename radprofT, typename eigenimT>
312 void radprofim( radprofT & radprof,
313  eigenimT & im,
314  radprofT & rad,
315  bool subtract = false)
316 {
317  typedef typename eigenimT::Scalar floatT;
318 
319  int dim1 = im.cols();
320  int dim2 = im.rows();
321 
322  floatT mr = rad.maxCoeff();
323 
324  /* A vector of radvals will be sorted, then binned*/
325  std::vector<radval<floatT> > rv(dim1*dim2);
326 
327  for(int i=0;i<rv.size();++i)
328  {
329  rv[i].r = rad(i);
330  rv[i].v = im(i);
331  }
332 
333  sort(rv.begin(), rv.end(), radvalRadComp<floatT>());
334 
335  /*Now bin*/
336  floatT dr = 1;
337  floatT r0 = 0;
338  floatT r1 = dr;
339  int i1=0, i2, n;
340 
341  floatT med;
342 
343  std::vector<double> med_r, med_v;
344  while(r1 < mr)
345  {
346  while(rv[i1].r < r0) ++i1;
347  i2 = i1;
348  while(rv[i2].r <= r1) ++i2;
349 
350  n = 0.5*(i2-i1);
351 
352  std::nth_element(rv.begin()+i1, rv.begin()+i1+n, rv.begin()+i2, radvalValComp<floatT>());
353 
354  med = (rv.begin()+i1+n)->v;
355 
356  med_r.push_back(.5*(r0+r1));
357  med_v.push_back(med);
358  i1 = i2;
359  r0 += dr;
360  r1 += dr;
361  }
362 
363  /* And finally, interpolate onto the radius image */
364  radprof.resize(dim1, dim2);
365  gslInterpolator interp(gsl_interp_linear, med_r, med_v);
366 
367  for(int i=0;i<dim1;++i)
368  {
369  for(int j=0;j<dim2;++j)
370  {
371  radprof(i,j) = interp.interpolate( ((double) rad(i,j)) );
372  if(subtract) im(i,j) -= radprof(i,j);
373  }
374  }
