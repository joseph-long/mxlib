247 template<typename realT>
248 int psdFilter<realT>::filter( realArrayT & noise )
249 {
250  //Make noise a complex number
251  for(int ii=0;ii<noise.rows();++ii)
252  {
253  for(int jj=0; jj<noise.cols(); ++jj)
254  {
255  _ftWork(ii,jj) = complexT(noise(ii,jj),0);
256  }
257  }
258 
259  //Transform complex noise to Fourier domain.
260  fft_fwd(_ftWork.data(), _ftWork.data() );
261 
262  //Apply the filter.
263  _ftWork *= *_psdSqrt;
264 
265  fft_back(_ftWork.data(), _ftWork.data());
266 
267  //Now take the real part, and normalize.
268  noise = _ftWork.real()/(noise.rows()*noise.cols());
269 
270  return 0;
271 }
