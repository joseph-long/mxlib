27 ///Convert a numerical value to a string
28 /** The default version uses the stream library to convert. A specialization is provided to prevent conversion of std::string.
29  *
30  * The precision parameter is only used for floating point types. If the default precision==0 is passed, then the maximum useful precision is
31  * used, from the value of std::numeric_limits<typeT>::max_digits10.
32  *
33  * The width and pad template parameters can be used to set a fixed maximum width and a pad character.
34  *
35  * Examples:
36  *
37  * To convert a double precision number to string:
38  * \code
39  double d = 2.898434;
40  std::string str;
41  str = convertToString(d); //note that you will not normally need to specify <typeT>
42  * \endcode
43  *
44  * To produce a fixed width 0 padded string from an integer:
45  * \code
46  int i = 23;
47  std::string str;
48  str = convertToString<int, 5, '0'>(i); //result is "00023".
49  * \endcode
50  *
51  * \tparam typeT is the type of value to convert
52  * \tparam width specifies the maximum width, not including the '\0'
53  * \tparam pad specifies the pad character
54  *
55  * \returns a string representation of value
56  *
57  */
58 template< typename typeT, unsigned width=0, char pad=' '>
59 std::string convertToString( const typeT & value, ///< [in] the value of type typeT to be converted
60  int precision = 0 ///< [in] [optional] the precision (see http://www.cplusplus.com/reference/ios/ios_base/precision/) to use for floating point types.
61  )
62 {
63 
64  std::ostringstream convert;
65 
66  if( std::is_floating_point<typeT>::value )
67  {
68  if(precision == 0 )
69  {
70  convert.precision( std::numeric_limits<typeT>::max_digits10);
71  }
72  else
73  {
74  convert.precision(precision);
75  }
76  }
77 
78  convert << value;
79 
80  if(width == 0)
81  {
82  return convert.str();
83  }
84  else
85  {
86  std::string c = convert.str();
87 
88  if( c.length() > width )
89  {
90  c.erase( width, c.length()-width);
91  return c;
92  }
93 
94  return std::string( width-c.length(), pad) + c;
95  }

300 /** First looks for 0/1,f/t, or F/T in the first non-space character of str.
301  * otherwise uses convertFromString<int>.
302  *
303  * \returns the converted numerical value.

306 bool convertFromString<bool>(const std::string & str /**< [in] is the string to convert*/ )
307 {
308  char c = str[0];
309  int i=0;

313  if(c == '1' || c == 't' || c == 'T') return true;
314

420 /// Wrap a string by breaking it into smaller sized portions of a desired width
421 /** Whenever possible breaks at spaces. A single space is discarded at the break.
422  */
423 inline
424 int stringWrap( std::vector<std::string> & lines, ///< [out] each new entry contains a wrapped portion of the string. Not cleared, so can accumulate.
425  const std::string & str, ///< [in] the string to wrap
426  int width ///< [in] the maximum width of the output strings
427  )
428 {
429  int L = str.length();
430 
431  if(L == 0) return 0;
432  int startPos, tmpPos, endPos;
433 
434  bool done = false;
435 
436  startPos = 0;
437 
438  while( !done )
439  {
440  if(startPos == L) --startPos; //Just to make sure
441 
442  endPos = startPos + width;
443 
444  if(endPos >= L)
445  {
446  lines.push_back( str.substr( startPos, L-startPos ));
447  done = true;
448  break;
449  }
450 
451  //Backup to nearest space
452  tmpPos = endPos;
453  while( !isspace(str[tmpPos]) && tmpPos >= startPos ) --tmpPos;
454 
455  //If we aren't at the beginning (i.e. splitting consecutive characters) we use new end position
456  if(tmpPos > startPos) endPos = tmpPos;
457 
458 
459  lines.push_back( str.substr( startPos, endPos-startPos) );
460 
461 
462  startPos = endPos;
463 
464  //Clear 1 space
465  if( str[startPos] == ' ') ++startPos;
466  }
467 
468  return 0;
469 }
