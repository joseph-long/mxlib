22 ///Fill in a vector with a regularly spaced scale
23 /** Fills in the vector with a 0....N-1 scale. The spacing of the points
24  * can be changed with the scale parameter, and the starting point can be
25  * changed with the offset.
26  *
27  * \tparam vectorT is a std::vector type.
28  */
29 template<typename vectorT>
30 void vectorScale( vectorT & vec, ///< [out] the vector to fill in, can be pre-allocated or not
31  size_t N = 0, ///< [in] [optional] if specified > 0, then vec is resize()-ed. Default is 0, and vec is not resize()-ed.
32  typename vectorT::value_type scale = 0, ///< [in] [optional] if specified !=0, then the points are spaced by this value. Default spacing is 1.
33  typename vectorT::value_type offset = 0 ///< [in] [optional] if specified !=0, then the starting point of the scale is this value.
34  )
35 {
36  if(scale == 0) scale = 1.0;

38  if(N > 0) vec.resize(N);
39

90 /** Returns the center element if vec has an odd number of elements. Returns the mean of the center 2 elements if vec has
91  * an even number of elements.
92  *
93  * \returns the median of vec
94  *
95  * \tparam vectorT the std::vector type of vec
96  *
97  */
98 template<typename vectorT>
99 typename vectorT::value_type vectorMedianInPlace(vectorT & vec /**< [in] the vector, is altered by std::nth_element*/)
100 {
101  typename vectorT::value_type med;
102 
103  int n = 0.5*vec.size();
104 
105  std::nth_element(vec.begin(), vec.begin()+n, vec.end());
106 
107  med = vec[n];
108 
109  //Average two points if even number of points

111  {
112  med = 0.5*(med + *std::max_element(vec.begin(), vec.begin()+n));
113  }
114 
115  return med;

194 ///Calculate the sigma-clipped mean of a vector
195 /** Performas sigma-clipping relative to the median, removing any values with deviation from the median > sigma.
196  * Continues until either no values are removed, or maxPasses iterations. If maxPasses == 0, then it is ignored.
197  *
198  * \returns the sigma clipped mean of vec
199  *
200  * \tparam vectorT the std::vector type of vec
201  *
202  */
203 template<typename vectorT>
204 typename vectorT::value_type vectorSigmaMean( const vectorT & vec, ///< [in] the vector (unaltered)
205  const vectorT * weights, ///< [in] [optional] the weights (unaltered)
206  typename vectorT::value_type sigma, ///< [in] the standard deviation threshold to apply.
207  int & maxPasses ///< [in/out] [optional] the maximum number of sigma-clipping passes. Set to actual number of passes on return.
208  )
209 {
210  vectorT work, wwork;
211 
212  typename vectorT::value_type med, var, Vsig, dev;
213 
214 
215  bool doWeight = false;
216  if( weights )
217  {
218  if( weights->size() == vec.size()) doWeight = true;
219  }
220 
221  Vsig = sigma*sigma;
222 
223  med = vectorMedian(vec, &work);
224  var = vectorVariance( work, med );
225 
226  int nclip;
227  int passes = 0;
228 
229 
230  //If weighting, have to synchronize work with weights since work will be
231  //partially sorted by median.
232  if(doWeight)
233  {
234  wwork.resize(vec.size());
235  for(int i=0; i< vec.size(); ++i)
236  {
237  work[i] = vec[i];
238  wwork[i] = (*weights)[i];
239  }
240  }
241 
242 
243  while(passes < maxPasses || maxPasses == 0)
244  {
245  ++passes;
246 
247  nclip = 0;
248 
249  for(size_t i=0; i<work.size(); ++i)
250  {
251  dev = pow(work[i] - med,2)/var;
252  if( dev > Vsig )
253  {
254  work.erase(work.begin()+i);
255 
256  if(doWeight) wwork.erase(wwork.begin() + i);
257 
258  --i;
259  ++nclip;
260  }
261  }
262 
263  if(nclip == 0) break;
264  med = vectorMedian(work);
265  var = vectorVariance( work, med );
266  }
267 
268  maxPasses = passes;
269 
270  if(doWeight)
271  {
272  return vectorMean(work, wwork);
273  }
274  else
275  {
276  return vectorMean(work);
277  }
278 }
