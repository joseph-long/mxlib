70 /* Strip whitespace chars off end of given string, in place. Return s. */
71 static char* rstrip(char* s)
72 {
73  char* p = s + strlen(s);
74  while (p > s && isspace((unsigned char)(*--p)))
75  *p = '\0';
76  return s;
77 }

79 /* Return pointer to first non-whitespace char in given string. */
80 static char* lskip(const char* s)
81 {
82  while (*s && isspace((unsigned char)(*s)))
83  s++;
84  return (char*)s;
85 }
86 
87 /* Return pointer to first char c or ';' comment in given string, or pointer to
88  null at end of string if neither found. ';' must be prefixed by a whitespace
89  character to register as a comment. */
90 static char* find_char_or_comment(const char* s, char c)
91 {
92  int was_whitespace = 0;
93  while (*s && *s != c && !(was_whitespace && *s == INI_COMMENT_CHAR)) {
94  was_whitespace = isspace((unsigned char)(*s));
95  s++;
96  }
97  return (char*)s;
98 }

100 /* Version of strncpy that ensures dest (size bytes) is null-terminated. */
101 static char* strncpy0(char* dest, const char* src, size_t size)
102 {
103  strncpy(dest, src, size);
104  dest[size - 1] = '\0';
105  return dest;
106 }

108 /// Parse ini file
109 /** Same as ini_parse(), but takes a FILE* instead of filename. This doesn't
110  close the file when it's finished -- the caller must do that. */
111 inline
112 int ini_parse_file(FILE* file,
113  int (*handler)(void*, const char*, const char*,
114  const char*),
115  void* user)
116 {
117  /* Uses a fair bit of stack (use heap instead if you need to) */
118 #if INI_USE_STACK
119  char line[INI_MAX_LINE];
120 #else
121  char* line;
122 #endif
123  char section[MAX_SECTION] = "";
124  char prev_name[MAX_NAME] = "";
125 
126  char* start;
127  char* end;
128  char* name;
129  char* value;
130  int lineno = 0;
131  int error = 0;
132 
133 #if !INI_USE_STACK
134  line = (char*)malloc(INI_MAX_LINE);
135  if (!line) {
136  return -2;
137  }
138 #endif
139 
140  /* Scan through file line by line */
141  while (fgets(line, INI_MAX_LINE, file) != NULL) {
142  lineno++;
143 
144  start = line;
145 #if INI_ALLOW_BOM
146  if (lineno == 1 && (unsigned char)start[0] == 0xEF &&
147  (unsigned char)start[1] == 0xBB &&
148  (unsigned char)start[2] == 0xBF) {
149  start += 3;
150  }
151 #endif
152  start = lskip(rstrip(start));
153 
154  if (*start == INI_COMMENT_CHAR || *start == '#') {
155  /* Per Python ConfigParser, allow '#' comments at start of line */
156  }
157 #if INI_ALLOW_MULTILINE
158  else if (*prev_name && *start && start > line) {
159  /* Non-black line with leading whitespace, treat as continuation
160  of previous name's value (as per Python ConfigParser). */
161  if (!handler(user, section, prev_name, start) && !error)
162  error = lineno;
163  }
164 #endif
165  else if (*start == '[') {
166  /* A "[section]" line */
167  end = find_char_or_comment(start + 1, ']');
168  if (*end == ']') {
169  *end = '\0';
170  strncpy0(section, start + 1, sizeof(section));
171  *prev_name = '\0';
172  }
173  else if (!error) {
174  /* No ']' found on section line */
175  error = lineno;
176  }
177  }
178  else if (*start && *start != INI_COMMENT_CHAR) {
179  /* Not a comment, must be a name[=:]value pair */
180  end = find_char_or_comment(start, '=');
181  if (*end != '=') {
182  end = find_char_or_comment(start, ':');
183  }
184  if (*end == '=' || *end == ':') {
185  *end = '\0';
186  name = rstrip(start);
187  value = lskip(end + 1);
188  end = find_char_or_comment(value, '\0');
189  if (*end == INI_COMMENT_CHAR)
190  *end = '\0';
191  rstrip(value);
192 
193  /* Valid name[=:]value pair found, call handler */
194  strncpy0(prev_name, name, sizeof(prev_name));
195  if (!handler(user, section, name, value) && !error)
196  error = lineno;
197  }
198  else if (!error) {
199  /* No '=' or ':' found on name[=:]value line */
200  error = lineno;
201  }
202  }
203 
204 #if INI_STOP_ON_FIRST_ERROR
205  if (error)
206  break;
207 #endif
208  }
209 
210 #if !INI_USE_STACK
211  free(line);
212 #endif
213 
214  return error;
215 }

217 /// Parse given INI-style file.
218 /** May have [section]s, name=value pairs
219  (whitespace stripped), and comments. By default comments start with ';' (semicolon)
220  but this can be configured with the INI_COMMENT_CHAR macro. Section
221  is "" if name=value pair parsed before any section heading. name:value
222  pairs are also supported as a concession to Python's ConfigParser.
223 
224  For each name=value pair parsed, call handler function with given user
225  pointer as well as section, name, and value (data only valid for duration
226  of handler call). Handler should return nonzero on success, zero on error.
227 
228  Returns 0 on success, line number of first error on parse error (doesn't
229  stop on first error), -1 on file open error, or -2 on memory allocation
230  error (only when INI_USE_STACK is zero).
231 */
232 inline
233 int ini_parse(const char* filename,
234  int (*handler)(void*, const char*, const char*, const char*),
235  void* user)
236 {
237  FILE* file;
238  int error;
239 
240  file = fopen(filename, "r");
241  if (!file)
242  return -1;
243  error = ini_parse_file(file, handler, user);
244  fclose(file);
245  return error;
246 }
