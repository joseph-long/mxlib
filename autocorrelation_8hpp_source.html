32 void autocorrelation( T * ac,
33  size_t Nac,
34  T * sig,
35  size_t Nsig )
36 {
37 
38  #pragma omp parallel for
39  for(int i=0; i< Nac; ++i)
40  {
41  ac[i] = 0;
42 
43  for(int j=0; j<Nsig-i; ++j)
44  {
45  ac[i] += sig[j]*sig[j+i];
46  }
47  }
48 
49  T norm = ac[0];
50  for(int i=0; i<Nac; ++i) ac[i] /= norm;
51 }
