92 /** Spawns a gnuplot sesssion and communicates with it.
93  * \ingroup plotting
94  *
95  * \todo Use mxError for error reporting.
96  *
97  * An example of using gnuPlot to plot data from a file:
98  * \code
99  * gnuPlot gp; //This automatically connects, and is now ready to plot.
100  *
101  * gp.plot("my_data_file.dat", "u 1:2 w l"); //Plots the data in the file
102  *
103  * std::cout << gp.getResponse("show terminal") << "\n"; //Sends the command to gnuplot, gets the response.
104  * \endcode
105  *
106  * Error checking and reporting is not straightforward since gnuplot does not return a result if there is no error. If there is an error,
107  * it can take several hundred milliseconds for a response to be available. So for a typical
108  * plot command, say, one does not want to try to read the stderr ouput just in case as this will just time out with a delay.
109  *
110  * For cases where a response is expected, the following example shows how to check for errors.
111  * \code
112  * errno = 0;
113  * std::string response = gp.getResponse("show terminal"); //Waits up to 0.5 seconds, or a user specified timeout.
114  *
115  * if(response == "") //This indicates some error occurred, or a timeout.
116  * {
117  * if(gp.gpError())
118  * {
119  * std::cerr << "gnuplot returned error:\n" << gp.gpErrorMsg() << "\n";
120  * }
121  * else if(errno)
122  * {
123  * perror("error getting response: ");
124  * }
125  * else
126  * {
127  * std::cerr << "timed out\n";
128  * }
129  * }
130  * else
131  * {
132  * std::cout << response << "\n";
133  * }
134  *
135  * \endcode

493 int gnuPlot::connect()
494 {
495 
496  if(_connected) return 0;
497 
498  srand( (unsigned long) this); //We don't need random, just different from all other instances in this process
499 
500 
501  char tmpStr[MX_GP_FNAME_SZ];
502 
503  snprintf(tmpStr, MX_GP_FNAME_SZ, "/gperr%d_%d", getpid(), rand());
504 
505  _errFName = _errLocation + tmpStr;
506 
507 
508  std::string comm = "gnuplot -persist 2>";
509  comm += _errFName;
510 
511  errno = 0;
512  _pipeH = popen(comm.c_str(), "w");
513 
514  if(_pipeH == NULL)
515  {
516  if(errno)
517  {
518  mxPError("gnuPlot", errno, "Error starting the gnuplot program with popen.");
519  }
520  else
521  {
522  mxError("gnuPlot", MXE_PROCERR, "Error starting the gnuplot program with popen.");
523  }
524 
525  return -1;
526  }
527 
528 
529 
530  usleep(MX_GP_FC_TIME);//allow file creation to finish before opening stderr file
531 
532  errno = 0;
533  _errFD = open(_errFName.c_str(), O_RDONLY);
534 
535  int n =0;
536  while(_errFD <= 0 && n < MX_GP_FC_RETRIES)
537  {
538  //First we try again after sleeping again.
539  usleep(MX_GP_FC_TIME);//allow file creation to finish before opening stderr file
540  _errFD = open(_errFName.c_str(), O_RDONLY);
541  ++n;
542  }
543 
544  if(_errFD <= 0)
545  {
546  mxPError("gnuPlot", errno, "gnuPlot failed to open stderr file: ");
547  return -1;
548  }
549 
550  _connected = 1;
551 }

579 inline
580 int gnuPlot::checkResponse(std::string & response, double timeout)
581 {
582  _gpError = false;
583  if(_errFD)
584  {
585  char errstr[1024];
586  int rv = 0;
587 
588  double t0 = gp_get_curr_time();
589 
590  errno = 0;
591  rv = read(_errFD, errstr, 1024);
592 
593  while(rv == 0 && gp_get_curr_time() - t0 <= timeout)
594  {
595  usleep(10); //Give up the thread
596  rv = read(_errFD, errstr, 1024);
597  }
598 
599  if(rv < 0)
600  {
601  mxPError("gnuPlot", errno, "Occurred while reading gnuplot stderr");
602 
603  response = "";
604  return -1;
605  }
606 
607  if(rv == 0)
608  {
609  mxError("gnuPlot", MXE_TIMEOUT, "Timed out while reading from gnuplot stderr");
610  response = "";
611  return 0;
612  }
613 
614  errstr[rv] = '\0';
615 
616  response = errstr;
617 
618  bool done = false;
619  if(response.length() > 1)
620  {
621  if(response[response.length()-1] == '\n' && response[response.length()-2] == '\n') done = true;
622  }
623 
624 
625  while( !done && gp_get_curr_time() - t0 < timeout)
626  {
627  usleep(10); //Give up the thread
628  rv = read(_errFD, errstr, 1024);
629 
630  if(rv > 0)
631  {
632  errstr[rv] = '\0';
633  response += errstr;
634 
635  if(response.length() > 1)
636  {
637  if(response[response.length()-1] == '\n' && response[response.length()-2] == '\n') done = true;
638  }
639  }
640  }
641 
642  int first =0;
643  while( isspace(response[first]) && first < response.length()-1) ++first;
644  response.erase(0, first);
645 
646 
647  int last = response.length()-1;
648  while( response[last] == '\n' && last > 0) --last;
649  if(response[last] != '\n') ++last;
650 
651  response.erase(last);
652 
653 
654  if(response.length() > 8)
655  {
656  if(response.substr(0,8) == "gnuplot>")
657  {
658  _gpError = true;
659  _gpErrorMsg = response;
660 
661  mxError("gnuPlot", MXE_GNUPLOTERR, "gnuplot says:\n" + response);
662 
663  response = "";
664  return -1;
665  }
666  }
667 
668 
669  return 0;
670  }
671 
672  //If errFD is not open for some reason
673  std::cerr << "gnuplot stderr is not open\n";
674 
675 
676 
677  return 1;
678 
679

869 int gnuPlot::circle( realT xcen, realT ycen, realT radius, const std::string & modifiers, const std::string & title, realT npoints)
870 {
871  int halfCirc = (0.5 * 2.0 * pi<realT>() * radius * npoints + 0.5);
872 
873  std::vector<realT> xpts(2*halfCirc), ypts(2*halfCirc);
874 
875  realT x, y;
876 
877  for(int i=0; i < halfCirc; ++i)
878  {
879  x = radius * cos(( (realT) i) / ( (realT) halfCirc) * pi<realT>());
880 
881  y = sqrt( radius*radius - x*x);
882 
883  xpts[i] = xcen + x;
884  ypts[i] = ycen + y;
885 
886  xpts[halfCirc + i] = xcen - x;
887  ypts[halfCirc + i] = ycen - y;
888  }
889 
890  return plot(xpts, ypts, modifiers, title);
891 }
