283 void lyotCoronagraph<_realT, _fpmaskFloatT>::applyApodizer( complexFieldT &pupilPlane )
284 {
285  int sz = pupilApodizer.rows();
286 
287  realT w = 0.5*(sz - 1.0);
288 
289  realT xc = 0.5*(pupilPlane.rows() - 1);
290  realT yc = 0.5*(pupilPlane.cols() - 1);
291 
292  for(int i=0; i<pupilPlane.rows(); ++i)
293  {
294  for(int j=0; j<pupilPlane.cols(); ++j)
295  {
296  if(i >= xc-w && i <= xc+w && j >= yc-w && j <= yc+w) pupilPlane( i, j) *= pupilApodizer(i-(xc-w),j-(yc-w));
297  else pupilPlane(i,j) *= 0;
298  }
299  }

383 template<typename _realT, typename _fpmaskFloatT>
384 void lyotCoronagraph<_realT, _fpmaskFloatT>::optimizeAPLCMC( imageT & geomPupil,
385  realT fpmRadPix,
386  realT relTol,
387  realT absTol,
388  int maxIter,
389  const std::string & cname )
390 {
391  complexFieldT focalPlane, pupilPlane;
392 
393  pupilPlane.resize(_wfSz, _wfSz);
394  focalPlane.resize(_wfSz, _wfSz);
395 
396  mx::imagingArray<realT,mx::fftwAllocator<realT>, 0> mask(_wfSz, _wfSz);
397  mx::wfp::circularPupil( mask, 0., fpmRadPix);
398 
399  //Initialize pupilImage
400  mx::imagingArray<realT,mx::fftwAllocator<realT>, 0> pupilImage(_wfSz, _wfSz);
401  pupilImage.setZero();
402 
403  int gpLLi = 0.5*(_wfSz-1) - 0.5*(geomPupil.rows()-1);
404  int gpLLj = 0.5*(_wfSz-1) - 0.5*(geomPupil.cols()-1);
405 
406  int gpURi = gpLLi + geomPupil.rows();
407  int gpURj = gpLLj + geomPupil.cols();
408 
409  for(int i=0;i<geomPupil.rows();++i)
410  {
411  for(int j=0;j< geomPupil.cols();++j)
412  {
413  pupilImage(gpLLi + i, gpLLj + j) = geomPupil(i,j);
414  }
415  }
416 
417  realT lastLambdaA, LambdaA;
418 
419  lastLambdaA = 1;
420  int n;
421  std::string reason;
422  for(n=0; n< maxIter; ++n)
423  {
424  mx::wfp::makeComplexPupil(pupilPlane, pupilImage, _wfSz);
425 
426  fi.propagatePupilToFocal(focalPlane, pupilPlane);
427 
428  for(int i=0; i < _wfSz; ++i)
429  {
430  for(int j=0;j < _wfSz; ++j)
431  {
432  focalPlane(i,j) *= mask(i,j);
433  }
434  }
435 
436  fi.propagateFocalToPupil(pupilPlane, focalPlane);
437 
438  for(int i=0; i< _wfSz; ++i) for(int j=0;j< _wfSz; ++j) pupilImage(i,j) = abs(pupilPlane(i,j));
439 
440 
441  LambdaA = 0;
442  for(int i=0; i< _wfSz; ++i)
443  {
444  for(int j=0;j< _wfSz; ++j)
445  {
446  if( i >= gpLLi && i < gpURi && j >= gpLLj && j < gpURj)
447  {
448  pupilImage(i,j) *= geomPupil(i-gpLLi,j-gpLLj);
449  if(pupilImage(i,j) > LambdaA) LambdaA = pupilImage(i,j);
450  }
451  else
452  {
453  pupilImage(i,j) = 0;
454  }
455  }
456  }
457  //LambdaA = 1.0/LambdaA;
458  for(int i=0; i< _wfSz; ++i) for(int j=0;j<_wfSz; ++j) pupilImage(i,j) /= LambdaA;
459 
460  std::cout << n << " " << LambdaA << "\n";
461  if( fabs(LambdaA - lastLambdaA) < absTol)
462  {
463  std::cout << "Converged on absTol.\n";
464  reason = "absTol";
465  break;
466  }
467  if( fabs((LambdaA - lastLambdaA)/lastLambdaA) < relTol)
468  {
469  std::cout << "Converged on relTol.\n";
470  reason = "relTol";
471  break;
472  }
473 
474  if(n == maxIter - 1)
475  {
476  std::cout << "maxIter reached.\n";
477  reason = "maxIter";
478  }
479 
480  lastLambdaA = LambdaA;
481  }
482  if(LambdaA > 1) LambdaA = 1;
483 
484  std::cout << "LambdaA: = " << LambdaA << "\n";
485 
486  realT trans = 1.0 - 1.0/LambdaA;
487 
488  int pupSize = geomPupil.rows();
489 
490  pupilApodizer.resize(pupSize, pupSize);
491 
492  extractBlock(pupilApodizer, 0, pupSize, 0, pupSize, pupilImage, 0.5*( (pupilImage.rows()-1) - (pupSize-1)), 0.5*( (pupilImage.rows()-1) - (pupSize-1)));
493 
494  makeFocalMask(fpmRadPix, trans, pupSize);
495  maskSource = 2;
496 
497  lyotStop = geomPupil;
498 
499 
500  fitsHeader head;
501 
502  head.append("", fitsCommentType(), "----------------------------------------");
503  head.append("", fitsCommentType(), "lyotCoronagraph optimization Parameters:");
504  head.append("", fitsCommentType(), "----------------------------------------");
505  head.append<int>("WFSZ", _wfSz, "Size of wavefront used for FFTs (pixels)");

507  head.append<realT>("ABSTOL", absTol , "input absolute tolerance");
508  head.append<realT>("RELTOL", relTol , "input relative tolerance");

512  head.append<realT>("FPMTRANS", trans, "transmission of FPM");
