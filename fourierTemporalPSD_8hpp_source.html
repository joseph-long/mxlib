400 template<typename realT, typename aosysT>
401 realT fourierTemporalPSD<realT, aosysT>::fastestPeak( int m,
402  int n )
403 {
404  realT ku, kv, vu,vv;
405 
406  ku = ( (realT) m / _aosys->D());
407  kv = ( (realT) n / _aosys->D());
408 
409  realT f, fmax = 0;
410 
411  for(int i=0; i< _aosys->atm.n_layers(); ++i)
412  {
413  vu = _aosys->atm.layer_v_wind(i) * cos(_aosys->atm.layer_dir(i));
414  vv = _aosys->atm.layer_v_wind(i) * sin(_aosys->atm.layer_dir(i));
415 
416  f = fabs(ku*vu + kv*vv);
417  if(f > fmax) fmax = f;
418  }
419 
420  return fmax;
421 }

774 template<typename realT, typename aosysT>
775 realT F_basic (realT kv, void * params)
776 {
777  fourierTemporalPSD<realT, aosysT> * Fp = (fourierTemporalPSD<realT, aosysT> *) params;
778 
779  realT f = Fp->_f;
780  realT v_wind = Fp->_aosys->atm.layer_v_wind(Fp->_layer_i);
781 
782  //realT r_0 = Fp->_aosys->atm.r_0();
783  //realT L_0 = Fp->_aosys->atm.L_0();
784  realT D = Fp->_aosys->D();
785  realT m = Fp->_m;
786  realT n = Fp->_n;
787  int p = Fp->_p;
788 
789  realT ku = f/v_wind;
790 
791  realT kp = sqrt( pow(ku + m/D,2) + pow(kv + n/D,2) );
792  realT kpp = sqrt( pow(ku - m/D,2) + pow(kv - n/D,2) );
793 
794  realT Q1 = math::func::jinc(pi<realT>()*D*kp);
795 
796  realT Q2 = math::func::jinc(pi<realT>()*D*kpp);
797 
798  realT Q = (Q1 + p*Q2);
799 
800  realT P = Fp->_aosys->psd(Fp->_aosys->atm, sqrt( pow(ku,2) + pow(kv,2)), 0, Fp->_aosys->lam_wfs());
801 
802  return P*Q*Q ;
803 }

844 template<typename realT, typename aosysT>
845 realT F_projMod (realT kv, void * params)
846 {

848 
849  realT f = Fp->_f;
850  realT v_wind = Fp->_aosys->atm.layer_v_wind(Fp->_layer_i);
851 
852  realT q_wind = Fp->_aosys->atm.layer_dir(Fp->_layer_i);
853 
854  //For rotating the basis
855  realT cq = cos(q_wind);
856  realT sq = sin(q_wind);
857 
858 
859  //realT r_0 = Fp->_aosys->atm.r_0();
860  //realT L_0 = Fp->_aosys->atm.L_0();
861  realT D = Fp->_aosys->D();

866  realT n;
867  int p, pp;// = Fp->_p;
868 
869  realT ku = f/v_wind;
870 
871  realT kp, kpp, km, kmp, Jp, Jpp, Jm, Jmp, QQ;
872 
873  QQ = 0;
874 
875  for(int i=0; i< Fp->_modeCoeffs.cols(); ++i)
876  {
877  if(Fp->_modeCoeffs(mode_i,i) == 0) continue;
878 
879  m = Fp->ms[i]*cq + Fp->ns[i]*sq;
880  n = -Fp->ms[i]*sq + Fp->ns[i]*cq;
881 
882  kp = sqrt( pow(ku + m/D,2) + pow(kv + n/D,2) );
883  km = sqrt( pow(ku - m/D,2) + pow(kv - n/D,2) );
884

887 
888  }
889

891 
892  realT sc;
893 
894  for(int i=0; i< N ; ++i)
895  {

899  Jm = Fp->Jms[i];

903

905  {
906  if( abs(Fp->_modeCoeffs(mode_i,j)) == 0) break; //continue;
907  sc = Fp->_modeCoeffs(mode_i,i)*Fp->_modeCoeffs(mode_i, j);

912 
913  pp = Fp->ps[j];
914 
915  if(p == pp)
916  {
917  QQ += 2*2*( Jp*Jpp + Jm*Jmp)*sc;
918  }
919  else
920  {
921  QQ += 2*2*( Jp*Jmp + Jm*Jpp)*sc;
922  }
923 
924  }
925  }

927 
928  //realT QQ = 2*(Jp*Jp + Jm*Jm);
