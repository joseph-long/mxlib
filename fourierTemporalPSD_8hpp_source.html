413 template<typename realT, typename aosysT>
414 realT fourierTemporalPSD<realT, aosysT>::fastestPeak( int m,
415  int n )
416 {
417  realT ku, kv, vu,vv;
418 
419  ku = ( (realT) m / _aosys->D());
420  kv = ( (realT) n / _aosys->D());
421 
422  realT f, fmax = 0;
423 
424  for(int i=0; i< _aosys->atm.n_layers(); ++i)
425  {
426  vu = _aosys->atm.layer_v_wind(i) * cos(_aosys->atm.layer_dir(i));
427  vv = _aosys->atm.layer_v_wind(i) * sin(_aosys->atm.layer_dir(i));
428 
429  f = fabs(ku*vu + kv*vv);
430  if(f > fmax) fmax = f;
431  }
432 
433  return fmax;
434 }

436 template<typename realT, typename aosysT>
437 int fourierTemporalPSD<realT, aosysT>::singleLayerPSD( std::vector<realT> &PSD,
438  std::vector<realT> &freq,
439  realT m,
440  realT n,
441  int layer_i,
442  int p,
443  realT fmax )
444 {
445 
446 
447  if(fmax == 0) fmax = freq[freq.size()-1];
448 
449  realT v_wind = _aosys->atm.layer_v_wind(layer_i);
450  realT q_wind = _aosys->atm.layer_dir(layer_i);
451  realT r_0 = _aosys->atm.r_0();
452  realT D = _aosys->D();
453 
454  //Rotate the basis
455  realT cq = cos(q_wind);
456  realT sq = sin(q_wind);
457 
458 
459  realT scale = 2*(1/v_wind); //Factor of 2 for negative frequencies.
460 
461  //We'll get the occasional failure to reach tolerance error, just ignore them all for now.
462  gsl_set_error_handler_off();
463 
464  //Create a local instance so that we're reentrant
465  fourierTemporalPSD<realT, aosysT> params(true);
466 
467  params._aosys = _aosys;
468  params._layer_i = layer_i;
469  params._m = m*cq + n*sq;
470  params._n = -m*sq + n*cq;
471  params._p = p;
472 
473 
474  params._mode_i = _mode_i;
475  params._modeCoeffs = _modeCoeffs;
476 
477 
478  realT result, error, result2;
479 
480  //Setup the GSL calculation
481  gsl_function func;
482  switch(_useBasis)
483  {
484  case basis::basic: //MXAO_FTPSD_BASIS_BASIC:
485  func.function = &F_basic<realT, aosysT>;
486  break;
487  case basis::modified: //MXAO_FTPSD_BASIS_MODIFIED:
488  func.function = &F_mod<realT, aosysT>;
489  break;
490  case basis::projected_basic: //MXAO_FTPSD_BASIS_PROJECTED_BASIC:
491  mxError("fourierTemporalPSD::singleLayerPSD", MXE_NOTIMPL, "Projected basic-basis modes are not implemented.");
492  break;
493  case basis::projected_modified:// MXAO_FTPSD_BASIS_PROJECTED_MODIFIED:
494  params.Jps = Jps;
495  params.Jms = Jms;
496  params.ms = ms;
497  params.ns = ns;
498  params.ps = ps;
499 
500 
501  func.function = &F_projMod<realT, aosysT>;
502 
503  break;
504  default:
505  mxError("fourierTemporalPSD::singleLayerPSD", MXE_INVALIDARG, "value of _useBasis isn not valid.");
506  return -1;
507  }
508 
509  func.params = &params;
510 
511 
512  //Here we only calculate up to fmax.
513  int i=0;
514  while(freq[i] <= fmax && i < freq.size())
515  {
516  params._f = freq[i];
517 
518  int ec = gsl_integration_qagi (&func, _absTol, _relTol, WSZ, params._w, &result, &error);
519 
520  if(ec == GSL_EDIVERGE)
521  {
522  std::cerr << "GSL_EDIVERGE:" << p << " " << freq[i] << " " << v_wind << " " << m << " " << n << " " << _m << " " << _n << "\n";
523  std::cerr << "ignoring . . .\n";
524  }
525 
526  PSD[i] = scale*result;
527 
528  ++i;
529  }
530 
531  //Now fill in from fmax to the actual max frequency with a -17/3 power law.
532  int j=i;
533 
534  //First average result for last 50.
535  PSD[j] = PSD[i-50] * pow( freq[i-50]/freq[j], seventeen_thirds<realT>());
536  for(int k=49; k> 0; --k)
537  {
538  PSD[j] += PSD[i-k] * pow( freq[i-k]/freq[j], seventeen_thirds<realT>());
539  }
540  PSD[j] /= 50.0;
541  ++j;
542  ++i;
543  while(j < freq.size())
544  {
545  PSD[j] = PSD[i-1] * pow( freq[i-1]/freq[j], seventeen_thirds<realT>());
546  ++j;
547  }
548 
549 
550 
551  return 0;
552 }
