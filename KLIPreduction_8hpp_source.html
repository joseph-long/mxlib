566  {
567  allidx[i].index = i;
568 
569  //CV is lower-triangular
570  if(i <= imno)
571  {
572  allidx[i].cvVal = CV(imno,i)/(sds[i]*sds[imno]);
573  }
574  else

576  allidx[i].cvVal = CV(i,imno)/(sds[i]*sds[imno]);

586

588  int j;

596  }
597  }

599  for(; j< Nims; ++j)

602  {
603  rotoff1 = j;
604  break;
605  }
606  }
607  }

609  {
610  rotoff0 = imno-dang;
611  if(rotoff0 < 0) rotoff0= 0;
612 
613  rotoff1 = imno+dang+1;
614  if(rotoff1 > Nims-1) rotoff1 = Nims;
615  }
616 
617  if(rotoff1-rotoff0 > 0)
618  {

631  }
