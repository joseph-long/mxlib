603 template<typename realT>
604 void aoAtmosphere<realT>::update_v_wind()
605 {
606 
607  if( _layer_v_wind.size() == 0 || _layer_Cn2.size() == 0)
608  {
609  _v_wind = 0;
610  _dir_wind = 0;
611  _v_wind_updated = true;
612  return;
613  }
614 
615  _v_wind = 0;
616 
617  realT s = 0;
618  realT c = 0;
619  realT t = 0;
620 
621  for(int i=0;i<_layer_Cn2.size(); ++i)
622  {
623  _v_wind += _layer_Cn2[i] * pow(_layer_v_wind[i], five_thirds<realT>() );
624  s += pow(_layer_v_wind[i], five_thirds<realT>())*sin(_layer_dir[i]) ;//pow( sin(_layer_dir[i]), five_thirds<realT>() );
625  c += pow(_layer_v_wind[i], five_thirds<realT>())*cos(_layer_dir[i]) ;//pow( cos(_layer_dir[i]), five_thirds<realT>() );
626  }
627 
628  _v_wind = pow(_v_wind, three_fifths<realT>());
629 
630  //_dir_wind = atan(pow(s, three_fifths<realT>()) / pow(c, three_fifths<realT>()));
631  _dir_wind = atan( s / c);
632  if(_dir_wind < 0) _dir_wind += pi<realT>();
633 
634  _v_wind_updated = true;
635 
636 }

772 {

775  realT tanZ = sinZ*secZ;
776  realT x0 = (n_air(lambda_wfs) - n_air(lambda_i)) * _H*tanZ*secZ;

783  }
