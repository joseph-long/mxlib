404 ///Form an estimate of the parameters of an elliptical Gaussian from a 2D image.
405 /** Note that this assumes that there is no constant value (i.e. it is zero-ed).
406  *
407  * \ingroup gaussian_peak_fit
408  */
409 template<typename realT>
410 int guessGauss2D_ang( realT & Ag, ///< [out] estimate of the peak
411  realT & xg, ///< [out] estimate of the x-coordinate of the peak
412  realT & yg, ///< [out] estimate of the y-coordinate of the peak
413  realT & xFWHM, ///< [out] estimate of the x-FWHM
414  realT & yFWHM, ///< [out] estimate of the y-FWHM
415  realT & angG, ///< [out] estimate of the angle of the ellipse
416  mx::improc::eigenImage<realT> & im, ///< [in] the image with an elliptical gaussian
417  realT maxWidth, ///< [in] the width of the box to search for the maximum
418  realT widthWidth, ///< [in] the radius of the circle to search for the widths
419  realT nAngs, ///< [in] the number of angles at which to search for the widths
420  realT xg0, ///< [in] an initial guess at the x-coordinate of the peak
421  realT yg0 ///< [in] an initial guess at the y-coordinate of the peak
422  )
423 {
424  xg = xg0;
425  yg = yg0;
426  Ag = im(xg, yg);
427  for(int i =0; i< 2*maxWidth+1; ++i)
428  {
429  for(int j=0; j< 2*maxWidth+1; ++j)
430  {
431  if( im( xg0 - maxWidth + i, yg0 - maxWidth + j) > Ag )
432  {
433  Ag = im( xg0 - maxWidth + i, yg0 - maxWidth + j);
434  xg = xg0 - maxWidth + i;
435  yg = yg0 - maxWidth + j;
436  }
437  }
438  }
439 
440  realT dAng = two_pi<realT>()/nAngs;
441  //std::vector<realT> dist(nAngs);
442 
443  realT c, s;
444 
445  realT maxD = 0;
446  int maxDidx = 0;
447  realT minD = widthWidth;
448  int minDidx = 0;
449 
450  for(int i=0; i < nAngs; ++i)
451  {
452  c = cos(i*dAng);
453  s = sin(i*dAng);
454 
455  for(int j=0; j < widthWidth; ++j)
456  {
457  if( im( xg + j*c, yg + j*s) <= 0.5*Ag )
458  {
459  //dist[i] = j;
460 
461  if(j > maxD)
462  {
463  maxD = j;
464  maxDidx = i;
465  }
466 
467  if(j < minD)
468  {
469  minD = j;
470  minDidx = i;
471  }
472  break;
473  }
474  }
475  }
476 
477  //Take minang and move it by 90 degrees
478  realT minang = fmod(minDidx * dAng - 0.5*pi<realT>(), pi<realT>());
479  if(minang < 0) minang = fmod(minang + two_pi<realT>(), pi<realT>());
480 
481  realT maxang = fmod(maxDidx * dAng, pi<realT>());
482 
483  //Now average
484  angG = 0.5*(minang + maxang);
485 
486  xFWHM = 2*maxD;
487  yFWHM = 2*minD;
488 
489  return 0; ///\returns 0 if successful
490 }
