23 /** The Jinc function is defined here as
24  * \f[
25  * Ji(x) = \frac{J_1(x)}{x}
26  * \f]
27  * where \f$ J_1 \f$ is the cylindrical bessel function of the first kind of order 1.
28  *
29  * Follows the technique in boost sinc_pi, using the Taylor series for small arguments. If x
30  * is smaller than \f$ \epsilon \f$, then it returns 1/2. If x is larger than \f$ \epsilon \f$ but smaler than \f$ \sqrt{\epsilon} \f$, then
31  * this function returns
32  * \f[
33  * Ji(x) \approx \frac{1}{2} - \frac{x^2}{16}.
34  * \f]
35  *
36  * \param x is the argument
37  *
38  * \returns the value of Ji(x)
39  *
40  * \tparam T is an floating point type
41  *
42  * \ingroup functions
43  */
44 template<typename T>
45 T jinc( const T & x)
46 {
47  T const taylor_0_bound = boost::math::tools::epsilon<T>();
48  T const taylor_2_bound = boost::math::tools::root_epsilon<T>();
49 
50  if( abs(x) > taylor_2_bound)
51  {
52  return boost::math::cyl_bessel_j<T>(1,x)/(x);
53  }
54  else
55  {
56  // approximation by taylor series in x at 0
57  T result = static_cast<T>(0.5);
58  if (abs(x) >= taylor_0_bound)
59  {
60  T x2 = x*x;
61 
62  // approximation by taylor series in x at 0 up to order 2
63  result -= x2/static_cast<T>(16);
64  }
65 
66  return result;
67  }
