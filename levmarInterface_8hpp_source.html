30 ///An interface to the levmar package
31 /** Requires a fitter class, which conforms to one of the following minimum specifications.
32  * To use the finite difference jacobian calculation:
33  * \code
34  * //This will cause the levmar_dif routine to be used
35  * template<typename _realT>
36  * struct dif_fitter
37  * {
38  * typedef _realT realT; //required
39  *
40  * static void func(realT *p, realT *hx, int m, int n, void *adata)
41  * {
42  * //do stuff here . . .
43  * }
44  * };
45  * \endcode
46  * If you wish to provide your own jacobian:
47  * \code
48  * //This will cause the levmar_der routine to be used
49  * template<typename _realT>
50  * struct der_fitter
51  * {
52  * typedef _realT realT; //required
53  *
54  * typdef bool hasJacobian; //this signals that jacf exists and should be used.
55  *
56  * static void func(realT *p, realT *hx, int m, int n, void *adata)
57  * {
58  * //do stuff here . . .
59  * }
60  *
61  * // This is the jacobian.
62  * static void jacf(realT *p, realT *j, int m, int n, void *adata)
63  * {
64  * //do stuff here . . .
65  * }
66  *
67  * };
68  *
69  * \endcode
70  *
71  * Note that if your fitter has a jacobian function (jacf), you must define
72  * \code
73  * typedef bool hasJacobian;
74  * \endcode
75  * for it to be used.
76  *
77  * \tparam fitterT a class with at least the minimum interface described above.
78  */

490 template<class fitterT>
491 void levmarInterface<fitterT>::allocate_work()
492 {
493  //Create function object to get allocation size, which depends on whether there is Jacobian.
494  levmar_allocate_size<fitterT> alloc_sz;
495 
496  if(work_sz < alloc_sz(m,n) || !work)
497  {
498  if(work) free(work);
499 
500  work_sz = alloc_sz(m,n);
501  work = (realT *) malloc( LM_DIF_WORKSZ(m,n) * sizeof(realT));
502  }
503 
504  //Allocate if covar is desired and unallocated.
505  if(getCovar)
506  {
507  if(covar_sz < m*m || !covar)
508  {
509  if(covar) free(covar);
510 
511  covar_sz = m*m;
512  covar = (realT *) malloc(covar_sz * sizeof(realT));
513  }
514  }
515  else
516  {
517  //If covar is not desired, de-allocate if allocated before.
518  if(covar) free(covar);
519  covar = 0;
520  }
521 }

717 ///Test whether a function type has a Jacobian function by testing whether it has a typedef of "hasJacobian"
718 /** Used for compile-time determination of whether the fitter has a Jacobian.
719  */
720 //This was taken directly from the example at http://en.wikipedia.org/wiki/Substitution_failure_is_not_an_error
721 template <typename T>
722 struct hasJacobian
723 {
724  // Types "yes" and "no" are guaranteed to have different sizes,
725  // specifically sizeof(yes) == 1 and sizeof(no) == 2.
726  typedef char yes[1];
727  typedef char no[2];
728 
729  template <typename fitterT>
730  static yes& test(typename fitterT::hasJacobian*);
731 
732  template <typename>
733  static no& test(...);
734 
735  /// If hasJacobian<fitterT>::value == true, then fitterT has a Jacobian and the appropriate levmar routines are used.
736  /// If ::value == false, then the numerical derivatives are calculated.
737  // If the "sizeof" of the result of calling test<T>(0) would be equal to sizeof(yes),
738  // the first overload worked and T has a nested type named "hasJacobian".
739  static const bool value = sizeof(test<T>(0)) == sizeof(yes);
740 };
