351 template<typename _realT, class _derotFunctObj>
352 int ADIobservation<_realT, _derotFunctObj>::injectFake( eigenImageT & fakePSF,
353  int image_i,
354  _realT PA,
355  _realT sep,
356  _realT contrast,
357  _realT scale)
358 {
359  //typedef Eigen::Array<realT, Eigen::Dynamic, Eigen::Dynamic> imT;
360 
361  //Check for correct sizing
362  if( (fakePSF.rows() < this->imc.rows() && fakePSF.cols() >= this->imc.cols()) ||
363  (fakePSF.rows() >= this->imc.rows() && fakePSF.cols() < this->imc.cols()))
364  {
365  throw mxException("mxlib:high contrast imaging", -1, "image wrong size", __FILE__, __LINE__, "fake PSF has different dimensions and can't be sized properly");
366  }
367 
368  //Check if fake needs to be padded out
369  if(fakePSF.rows() < this->imc.rows() && fakePSF.cols() < this->imc.cols())
370  {
371  eigenImageT pfake(this->imc.rows(), this->imc.cols());
372  padImage(pfake, fakePSF, 0.5*(this->imc.rows()-fakePSF.rows()), 0);
373  fakePSF = pfake;
374  }
375 
376  //Check if fake needs to be cut down
377  if(fakePSF.rows() > this->imc.rows() && fakePSF.cols() > this->imc.cols())
378  {
379  eigenImageT cfake(this->imc.rows(), this->imc.cols());
380  cutPaddedImage(cfake, fakePSF, 0.5*(fakePSF.rows() - this->imc.rows()));
381  fakePSF = cfake;
382  }
383 
384  /*** Now shift to the separation and PA, scale, apply contrast, and inject ***/
385  //allocate shifted fake psf
386  eigenImageT shiftFake(fakePSF.rows(), fakePSF.cols());
387 
388  realT ang, dx, dy;
389 
390  ang = dtor(-1*PA) + derotF.derotAngle(image_i);
391 
392  dx = sep * sin(ang);
393  dy = sep * cos(ang);
394 
395  imageShift(shiftFake, fakePSF, dx, dy, cubicConvolTransform<realT>());
396 
397  this->imc.image(image_i) = this->imc.image(image_i) + shiftFake*scale*contrast;
398 
399  return 0;
400 
401 }
