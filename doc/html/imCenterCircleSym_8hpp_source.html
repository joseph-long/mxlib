23 ///Find the center of the image of a point source using circular symmetry of the PSF.
24 /** Conducts a grid search through possible center positions. Finds the point with the minimum
25  * total variance around concentric circles about that point. The cost grid is fit with a Gaussian.

103  grid.resize( 2*(maxPix/dPix) + 1, 2*(maxPix/dPix) + 1);

106  #pragma omp parallel
107  {
108 #endif
109  int nRad = (maxRad - minRad)/dRad+1;
110  std::vector< std::vector<realT> > rads;
111  rads.resize( nRad );
112 
113  int N = 2*(maxPix/dPix)+1;
114 
115  realT startX, startY;
116 #ifdef ICCS_OMP
117  #pragma omp for
118 #endif
119  for(int i =0; i<N; ++i)
120  {
121  startX = x0 - maxPix + i*dPix;
122  for(int j=0; j<N;++j)
123  {
124  startY = y0 - maxPix + j*dPix;
125 
126  for(int k =0; k< rads.size(); ++k)
127  {
128  rads[k].clear();
129  }
130 
131  for( int ii = startX - maxRad; ii <= startX+maxRad; ++ii)
132  {
133  for( int jj = startY - maxRad; jj <= startY+maxRad; ++jj)
134  {
135 
136  if(doMask)
137  {
138  if( (*mask)(ii,jj) == 0) continue;
139  }
140 
141  realT rad;
142  rad = sqrt(pow( ii-startX, 2) + pow( jj-startY,2));
143 
144  if(rad < minRad || rad >= maxRad) continue;
145 
146  rads[ (rad-minRad)/dRad ].push_back( im(ii,jj) );
147  }//jj
148  }//ii
149 
150  grid(i,j) = 0;
151 
152  realT mean, var;
153  for(int k =0; k< rads.size(); ++k)
154  {
155  if(rads[k].size() <= 1) continue;
156  mean = vectorMean(rads[k]);
157  var = vectorVariance(rads[k], mean);
158 
159  grid(i,j) += var/pow(mean,2);
160  }
161  } //j
162  } //i
163 #ifdef ICCS_OMP
164  } //omp parallel
165 #endif
166  //-------- Now Smooth the grid, and multiply by -1 for fitting -------------
167 
168  if(smWidth > 0)
169  {
170  filterImage(smGrid, grid, gaussKernel<eigenImage<realT>,2>(10), 0);
171  }
172  else
173  {
174  smGrid = grid;
175  }
176 
177  smGrid *= -1;
178 
179 
180  //Get initial guess for the fit.
181  realT mx, mn, gx, gy;
182 
183  mx = smGrid.maxCoeff(&gx, &gy);
184  mn = smGrid.minCoeff();
185 
186 
187  //------------- Now fit the smoothed cost grid to a 2D elliptical Gaussian ------------
188  /** \todo This needs to be able to handle highly non-symmetric peaks much better.
189  */
190  fit.setArray(smGrid.data(), smGrid.rows(), smGrid.cols());
191  fit.setGuess(mn, mx-mn, gx, gy, guessWidth/dPix, guessWidth/dPix, 0);
192 
193  fit.fit();
194 
195  //The estimated enter of circular symmetry from the fit.
196  x = x0 - maxPix + fit.x0()*dPix;
197  y = y0 - maxPix + fit.y0()*dPix;
198 
199
