538 
539 template<typename realT>
540 std::complex<realT> clGainOpt<realT>::olXfer(int fi, complexT & H_dm, complexT & H_del, complexT & H_con)
541 {
542  if(_f[fi] <= 0)
543  {
544  H_dm = 0;
545  H_del = 0;
546  H_con = 0;
547  return 0;
548  }
549 
550  #ifdef PRECALC_TRIG
551  if(_fChanged)
552  {
553  int jmax = std::max(_a.size()+1, _b.size());
554 
555  _cs.resize(_f.size(), jmax);
556  _ss.resize(_f.size(), jmax);
557 
558  for(int i=0; i< _f.size(); ++i)
559  {
560  _cs(i,0) = 1.0;
561  _ss(i,0) = 0.0;
562 
563  for(int j = 1; j<jmax; ++j)
564  {
565  _cs(i,j) = cos(two_pi<realT>()*_f[i]*_Ti*realT(j));
566  _ss(i,j) = sin(two_pi<realT>()*_f[i]*_Ti*realT(j));
567  }
568  }
569 
570  _fChanged = false;
571  }
572  #endif
573 
574  if(_changed)
575  {
576  _H_dm.resize(_f.size());
577  _H_wfs.resize(_f.size());
578  _H_ma.resize(_f.size());
579  _H_del.resize(_f.size());
580  _H_con.resize(_f.size());
581 
582  int jmax = std::min(_a.size(), _b.size());
583 
584  //#pragma omp parallel for
585  for(int i=0; i<_f.size(); ++i)
586  {
587  if(_f[i] <=0 ) continue;
588 
589  complexT s = complexT(0.0, two_pi<realT>()*_f[i]);
590 
591  complexT expsT = exp(-s*_Ti);
592 
593  _H_dm[i] = (realT(1) - expsT)/(s*_Ti);
594 
595  _H_wfs[i] = _H_dm[i];
596 
597  _H_ma[i] = 1; //realT(1./_N)*(realT(1) - pow(expsT,_N))/(realT(1) - expsT);
598 
599  _H_del[i] = exp(-s*_tau);
600 
601  complexT FIR = complexT(_b[0],0);
602 
603  complexT IIR = complexT(0.0, 0.0);
604  for(int j = 1; j < jmax; ++j)
605  {
606  #ifdef PRECALC_TRIG
607  realT cs = _cs(i,j);
608  realT ss = _ss(i,j);
609  FIR += _b[j]*complexT(cs, -ss);
610  IIR += _a[j-1]*complexT(cs, -ss);
611  #else
612  complexT expZ = exp( -s*_Ti*realT(j));
613  FIR += _b[j]*expZ;
614  IIR += _a[j-1]*expZ;
615  #endif
616  }
617 
618  for(int jj=jmax; jj< _a.size()+1; ++jj)
619  {
620  #ifdef PRECALC_TRIG
621  realT cs = _cs(i,jj);
622  realT ss = _ss(i,jj);
623  IIR += _a[jj-1]*complexT(cs, -ss);
624  #else
625  complexT expZ = exp( -s*_Ti*realT(jj));
626  IIR += _a[jj-1]*expZ;
627  #endif
628  }
629 
630  for(int jj=jmax; jj<_b.size(); ++jj)
631  {
632  #ifdef PRECALC_TRIG
633  realT cs = _cs(i,jj);
634  realT ss = _ss(i,jj);
635  FIR += _b[jj]*complexT(cs, -ss);
636  #else
637  complexT expZ = exp( -s*_Ti*realT(jj));
638  FIR += _b[jj]*expZ;
639  #endif
640  }
641 
642  _H_con[i] = FIR/( realT(1.0) - IIR);
643 
644  }
645 
646  _changed = false;
647  }
648 
649  H_dm = _H_dm[fi];
650  H_del = _H_del[fi];//*_H_ma[fi];
651  H_con = _H_con[fi];
652 
653  return ( _H_dm[fi]*_H_wfs[fi]*_H_del[fi]*_H_con[fi]);
654 
655 }

716  std::cerr << "clVariance: Frequency grid and PSDs must be same size." << std::endl;
717  return -1;
718  }
719 
720  realT ETF, NTF, df;
721 
722  varErr = 0;
723  varNoise = 0;
724 
725  df = _f[1] - _f[0];
726 
727  for(int i=0; i< PSDerr.size(); ++i)
728  {
729  if(g == 0)
730  {
731  ETF = 1;
732  NTF = 0;
733  }
734  else
735  {
736  clTF2(ETF, NTF, i, g);
737  }
738  varErr += ETF*PSDerr[i]*df;
739  varNoise += NTF*PSDnoise[i]*df;
740  }
741 
742  return varErr + varNoise;
743 }
