54 ///Worker for the azimuthal integral (in phi) for the basic Fourier mode covariance.
55 /**
56  * \param phi the angle at which to evaluate the integrand
57  * \param params a pointer to a object of type fourierCovariance<floatT, aosyT>
58  *
59  * \tparam floatT a floating point type used for all calculations
60  * \tparam aosysT the type of the AO system structure
61  *
62  * \ingroup mxAOAnalytic
63  */
64 template<typename floatT, typename aosysT>
65 floatT phiInt_basic (floatT phi, void * params)
66 {
67 
68  fourierCovariance<floatT, aosysT> * Pp = (fourierCovariance<floatT, aosysT> *) params;
69 
70  floatT L0 = Pp->aosys->atm.L_0();
71  floatT D = Pp->aosys->D();
72 
73  int p = Pp->p;
74  floatT m = Pp->m;
75  floatT n = Pp->n;
76 
77  int pp = Pp->pp;
78  floatT mp = Pp->mp;
79  floatT np = Pp->np;
80 
81  floatT k = Pp->k;
82 
83  floatT cosp = cos(phi);
84  floatT sinp = sin(phi);
85 
86  /*** no prime ***/
87  floatT kmn_p, kmn_m;
88  floatT Ji_mn_p, Ji_mn_m;
89  floatT Q_mn;//, Qs_mn;
90 
91  kmn_p = sqrt( pow(k*cosp + m/D, 2) + pow(k*sinp + n/D, 2));
92  kmn_m = sqrt( pow(k*cosp - m/D, 2) + pow(k*sinp - n/D, 2));
93 
94  Ji_mn_p = math::func::jinc(pi<floatT>()*D*kmn_p);
95  Ji_mn_m = math::func::jinc(pi<floatT>()*D*kmn_m);
96 
97 
98  floatT N = 1./sqrt(0.5 + p*math::func::jinc(2*pi<floatT>()*sqrt(m*m+n*n)));
99 
100  Q_mn = N*(Ji_mn_p + p*Ji_mn_m);
101 
102  /*** primed ***/
103  floatT kmpnp_p, kmpnp_m;
104  floatT Ji_mpnp_p, Ji_mpnp_m;
105  floatT Q_mpnp;//, Qs_mpnp;
106 
107  kmpnp_p = sqrt( pow(k*cosp + mp/D, 2) + pow(k*sinp + np/D, 2));
108  kmpnp_m = sqrt( pow(k*cosp - mp/D, 2) + pow(k*sinp - np/D, 2));
109 
110 
111  Ji_mpnp_p = math::func::jinc(pi<floatT>()*D*kmpnp_p);
112  Ji_mpnp_m = math::func::jinc(pi<floatT>()*D*kmpnp_m);
113 
114  floatT Np = 1./sqrt(0.5 + pp*math::func::jinc(2*pi<floatT>()*sqrt(mp*mp+np*np)));
115 
116  Q_mpnp = Np*(Ji_mpnp_p + pp*Ji_mpnp_m);
117 
118  floatT P = Pp->aosys->psd(Pp->aosys->atm, k); //vonKarmanPSD(k, D, L0, Pp->subPiston, Pp->subTipTilt);
119 
120  return P*k*(Q_mn*Q_mpnp);
121 }

163 
164 
165  /*** primed ***/
166  floatT kmpnp_p, kmpnp_m;
167  floatT Ji_mpnp_p, Ji_mpnp_m;
168  floatT Qc_mpnp, Qs_mpnp;
169 
170  kmpnp_p = sqrt( pow(k*cosp + mp/D, 2) + pow(k*sinp + np/D, 2));
171  kmpnp_m = sqrt( pow(k*cosp - mp/D, 2) + pow(k*sinp - np/D, 2));
172 
173 
174  Ji_mpnp_p = math::func::jinc(pi<floatT>()*D*kmpnp_p);
175  Ji_mpnp_m = math::func::jinc(pi<floatT>()*D*kmpnp_m);
176 
177  floatT QQ;
178 
179  if(p == pp)
180  {
181  QQ = 2.0*( Ji_mn_p*Ji_mpnp_p + Ji_mn_m*Ji_mpnp_m);
182  }
183  else
184  {
185  QQ = 2.0*( Ji_mn_p*Ji_mpnp_m + Ji_mn_m*Ji_mpnp_p);
186  }
187

194 ///Worker function for the radial integral in the covariance calculation
195 /**
196  * \param k the spatial frequency at which to evaluate the integrand
197  * \param params a pointer to a object of type fourierCovariance<floatT, aosyT>
198  *
199  * \tparam floatT a floating point type used for all calculations. As of Nov 2016 must be double due to gsl_integration.
200  * \tparam aosysT the type of the AO system structure
201  */
202 template<typename floatT, typename aosysT>
203 floatT kInt (floatT k, void * params)
204 {
205  fourierCovariance<floatT, aosysT> * Pp = (fourierCovariance<floatT, aosysT> *) params;
206 
207  floatT result, error;
208 
209  gsl_function func;
210 
211  //Here choose between basic and modified Fourier modes.
212  if(Pp->useBasic)
213  {
214  func.function = &phiInt_basic<floatT, aosysT>;
215  }
216  else
217  {
218  func.function = &phiInt_mod<floatT, aosysT>;
219  }
220  func.params = Pp;
221 
222  Pp->k = k;
223 
224  //Tolerances of 1e-4 seem to be all we can ask for
225  gsl_integration_qag(&func, 0., 2*pi<double>(), 1e-10, 1e-4, WSZ, GSL_INTEG_GAUSS21, Pp->phi_w, &result, &error);
226 
227 
228  return result;
229 }

249  floatT m;
